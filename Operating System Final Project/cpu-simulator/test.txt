1 SET 0 51       # sum = 0
# Loop start
2 ADDI 51 50     # sum = sum + i
3 SYSCALL PRN 51 # print the sum so far
4 ADD 50 -1      # i = i - 1
5 JIF 50 9       # if i <= 0, go to instruction 9 (done)
# Jump back to loop start
6 SET -1 52      # set negative value
7 JIF 52 2       # always jumps to 2
# This line should never execute
8 HLT
# Done
9 SYSCALL PRN 51 # print final sum
10 HLT           # halt the CPU
End Instruction Section
